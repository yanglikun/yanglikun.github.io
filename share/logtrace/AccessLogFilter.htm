1. public class AccessLogFilter implements Filter {
  3.     private static final Logger logger = LoggerFactory.getLogger("accessLog");
  5.     public static final String STR_IP = "ip";
  7.     public static final String STR_EQ = "=";
  9.     public static final String STR_AND = "&";
  11.     public static final String STR_USERPIN = "userPin";
  13.     // 截取参数的最大长度
  14.     protected int maxLength = 20;
  17.     @Override
  18.     public void init(FilterConfig filterConfig) throws ServletException {
  20.     }
  22.     public void doFilter(ServletRequest servletRequest, ServletResponse servletResponse, FilterChain chain)
  23.             throws IOException, ServletException {
  24.         HttpServletRequest request = (HttpServletRequest) servletRequest;
  25.         HttpServletResponse response = (HttpServletResponse) servletResponse;
  26.         clearThreadContext();
  27.         setThreadContext(request);
  28.         long startTime = System.currentTimeMillis();
  29.         try {
  30.             chain.doFilter(request, response);
  31.         } finally {
  32.             logAndClearThreadContext(request, (System.currentTimeMillis() - startTime));
  33.         }
  35.     }
  37.     private void logAndClearThreadContext(HttpServletRequest request, long executionTime) {
  38.         try {
  39.             StringBuilder message = new StringBuilder(500);
  40.             message.append("Controller:");
  41.             message.append(request.getRequestURI());
  42.             message.append("|Params:");
  43.             StringBuilder params = new StringBuilder();
  44.             Enumeration parameters = request.getParameterNames();
  45.             while (parameters.hasMoreElements()) {
  46.                 String param = (String) parameters.nextElement();
  47.                 String value = request.getParameter(param);
  48.                 params.append(param);
  49.                 params.append(STR_EQ);
  50.                 params.append(StringUtils.left(value, maxLength));
  51.                 params.append(STR_AND);
  52.             }
  53.             if (params.toString().length() > 0) {
  54.                 message.append(params.deleteCharAt(params.length() - 1));
  55.             }
  56.             message.append("|Spend=").append(executionTime);
  57.             // 记录日志
  58.             logger.info(message.toString());
  59.         } finally {
  60.             clearThreadContext();
  61.         }
  62.     }
  64.     private void setThreadContext(HttpServletRequest request) {
  65.         ThreadContext.put("invokeNo", UUIDUtils.getUUID());
  66.         String ip = getIp(request);
  67.         ThreadContext.put(AccessLogFilter.STR_IP, ip == null ? "" : ip);
  68.     }
  70.     private void clearLoginContext() {
  71.         try {
  72.             LoginContext.clear();
  73.         } catch (Exception e) {
  74.             logger.error("清除LoginContext异常.", e);
  75.         }
  76.     }
  78.     private void clearThreadContext() {
  79.         try {
  80.             ThreadContext.clearAll();
  81.         } catch (Exception e) {
  82.             logger.error("清除ThreadContext异常.", e);
  83.         }
  84.     }
  86.     public void destroy() {
  87.     }
  89.     public int getMaxLength() {
  90.         return maxLength;
  91.     }
  93.     public void setMaxLength(int maxLength) {
  94.         this.maxLength = maxLength;
  95.     }
  97.     public String getIp(HttpServletRequest request) {
  98.         String ip = request.getHeader("J-Forwarded-For");
  99.         if (StringUtils.isEmpty(ip) || "unknown".equalsIgnoreCase(ip)) {
  100.             ip = request.getHeader("J-Forwarded-For");
  101.         }
  102.         if (StringUtils.isEmpty(ip) || "unknown".equalsIgnoreCase(ip)) {
  103.             ip = request.getHeader("X-Forwarded-For");
  104.         }
  105.         if (StringUtils.isEmpty(ip) || "unknown".equalsIgnoreCase(ip)) {
  106.             ip = request.getHeader("Proxy-Client-IP");
  107.         }
  108.         if (StringUtils.isEmpty(ip) || "unknown".equalsIgnoreCase(ip)) {
  109.             ip = request.getHeader("WL-Proxy-Client-IP");
  110.         }
  111.         if (StringUtils.isEmpty(ip) || "unknown".equalsIgnoreCase(ip)) {
  112.             ip = request.getRemoteAddr();
  113.         }
  114.         return ip;
  115.     }
